    # Seata

    # 1.下载

    从官网下载 seata-server-1.4.2,解压缩到工作目录。

    # 2.修改配置

    文件名：registry.conf

    registry {
  type = "nacos"
  nacos {
    application = "seata-server"
    serverAddr = "127.0.0.1:8848"
    group = "SEATA_GROUP"
    namespace = "a667f2c4-d1f7-466b-80e7-2fa043135670"
    cluster = "default"
    username = "nacos"
    password = "nacos123456"
  }
}
config {
  type = "nacos"
  nacos {
    serverAddr = "127.0.0.1:8848"
    namespace = "a667f2c4-d1f7-466b-80e7-2fa043135670"
    group = "SEATA_GROUP"
    username = "nacos"
    password = "nacos123456"
    dataId = "seataServer.properties"
  }
}
